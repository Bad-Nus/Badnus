# 01
[例题](http://ybt.ssoier.cn:8088/problem_show.php?pid=1267)
```cpp
#include<bits/stdc++.h>
#define rep(u,x,y) for(int u=x;u<=y;++u)
#define per(u,x,y) for(int u=x;u>=y;--u)
using namespace std;
const int N=1001;
int n,m;
int dp[N],w[N],v[N];
int main(){
    cin>>n>>m;
    rep(i,1,n)cin>>w[i]>>v[i];
    rep(i,1,n)per(j,m,w[i])dp[j]=max(dp[j],dp[j-w[i]]+v[i]);
    cout<<dp[m];
    return 0;
}
```

# 二进制优化多重背包
[例题](http://ybt.ssoier.cn:8088/problem_show.php?pid=1269)
```cpp
#include<bits/stdc++.h>
#define rep(u,x,y) for(int u=x;u<=y;++u)
#define per(u,x,y) for(int u=x;u>=y;--u)
using namespace std;
typedef long long ll;
const int N=6001;
int n,m,x,y,z,c,idx;
ll w[N],v[N],dp[N];
int main(){
    cin>>n>>m;
    rep(i,1,n){
        cin>>x>>y>>z,c=1;
        while(z>c){
            w[++idx]=c*x,v[idx]=c*y;
            z-=c,c<<=1;
        }
        w[++idx]=z*x,v[idx]=z*y;
    }
    rep(i,1,idx)per(j,m,w[i])dp[j]=max(dp[j],dp[j-w[i]]+v[i]);
    cout<<dp[m];
    return 0;
}
```
