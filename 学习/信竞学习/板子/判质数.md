# 单判

```cpp
bool isp(int x){
	if(x==2||x==3)return 1;
	if(x==1||x%6!=5&&x%6!=1)return 0;
	int tmp=sqrt(x);
	for(int i=5;i<=tmp;i+=6)if(x%i==0||x%(i+2)==0)return 0;
	return 1;
}
```

# 欧拉筛

```cpp
bool npri[N];vector<int>pri;
void euler(){
	rep(i,2,n){
		if(!npri[i])pri.emplace_back(i);
		for(int k:pri){
			if(i*k>n)break;
			npri[i*k]=1;
			if(i%k==0)break;
		}
	}
}
```
