# 裸板

```cpp
const int N=1e6+1;
int n,k,a[N];
deque<int>q;
void getmin(){
	rep(i,1,n){
		while(q.size()&&a[q.back()]>=a[i])q.pop_back();
		q.push_back(i);
		while(q.front()+k<=i)q.pop_front();
		if(i>=k)cout<<a[q.front()]<<' ';
	}
}
void getmax(){
	rep(i,1,n){
		while(q.size()&&a[q.back()]<=a[i])q.pop_back();
		q.push_back(i);
		while(q.front()+k<=i)q.pop_front();
		if(i>=k)cout<<a[q.front()]<<' ';
	}
}
```
